class Solution {
   public int largestAltitude(int[] gain) {
       int runningAltitude = 0;
       int maxAltitude = 0;
       for (int i = 0; i < gain.length; i++) {
           runningAltitude += gain[i];
           if (runningAltitude > maxAltitude) {
               maxAltitude = runningAltitude;
           }
       }
       return maxAltitude;
   }
}
